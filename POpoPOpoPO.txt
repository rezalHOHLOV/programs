#include <stdio.h>
#include <math.h>
int main () {
    printf("5 + 3 = %d\n", 5 + 3);
    printf("10 - 4 = %d\n", 10 - 4);
    printf("6 * 7 = %d\n", 6 * 7);
    printf("15 / 3 = %d\n", 15 / 3);
    printf("17 %% 4 = %d\n", 17 % 4);
    printf("2 ** 5 = %.0f\n", pow(2, 5));
    return 0;
}
