#include <stdio.h>

int main() {
    int sum = 0;
    for (int i = 1; i <= 5; i++) {
        sum += i;
        printf("i = %d, сумма = %d\n", i, sum);
    }
    printf("\nТаблица умножения на 3:\n");
    for (int i = 1; i <= 10; i++) {
        printf("3 * %d = %d\n", i, 3 * i);
    }
    return 0;
}
